#arbitrary arguments
def courses(*CLASS):
    for subject in CLASS:
        print(subject)
courses("CPA","CISCO","OOSAD")
#using return
def courses(*CLASS):
    for subject in CLASS:
        return subject
print(courses("CPA","CISCO","OOSAD"))
#using keywards
def courses(**take):
    for key,value in take.items():
        print("{}:{}".format(key,value))

courses(first='CPA',second='CISCO',third='OOSAD')

#program to calculate area of a circle

    print('the area is' + str(area))
area_of_circle()
#area of a cyrinder
def area_of_a_cylinder():
    pi = 3.142
    r = int(input("enter the radius" + " "))
    h = int(input("enter the height" + " "))
    area = (pi*r*h)
    print("The area is" + str(area))
area_of_a_cylinder()
